Usage Example

// Initialize TooltipFlow
const tooltip = new TooltipFlow({
    defaultPosition: 'top',
    theme: 'light',
    animation: 'fade',
    interactive: false
});

// HTML Usage
<div data-tooltip="Tooltip content"
     data-tooltip-position="top"
     data-tooltip-theme="light">
    Hover me
</div>
